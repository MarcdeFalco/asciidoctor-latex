Overview

TeXPostProcess cleans up undesired transformations inside the TeX environment. Strings &ampp;, &gt;, &lt; are mapped back to &, >, < and \ is conserved.

def self.make_substitutions str
  str = str.gsub('\\\\', '@@')   # (1)
  matches = match_inline str
  if matches.count > 0
    str = make_substitutions_in_matches matches, str
  end
  matches = match_block str
  if matches.count > 0
    str = make_substitutions_in_matches matches, str
  end
  str = str.tr('@','\\')         # (2)
  str
end

def self.make_substitutions_in_matches matches, str
  matches.each do |m|
    m_str = m[0]
    m_transformed = make_substitutions1 m_str
    str = str.gsub(m_str,m_transformed)
  end
  str
end
